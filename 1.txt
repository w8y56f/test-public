session.doWork(connection -> {
    DatabaseMetaData metaData = connection.getMetaData();
    
    // 1. 先获取主键列名
    List<String> pkColumnNames = new ArrayList<>();
    try (ResultSet rsPk = metaData.getPrimaryKeys(null, null, tableName)) {
        while (rsPk.next()) {
            pkColumnNames.add(rsPk.getString("COLUMN_NAME"));
        }
    }

    // 2. 根据列名获取详细元数据 (DATA_TYPE 等)
    for (String pkColName : pkColumnNames) {
        try (ResultSet rsCol = metaData.getColumns(null, null, tableName, pkColName)) {
            if (rsCol.next()) {
                int dataType = rsCol.getInt("DATA_TYPE");
                String typeName = rsCol.getString("TYPE_NAME");
                int columnSize = rsCol.getInt("COLUMN_SIZE");
                boolean isNullable = (rsCol.getInt("NULLABLE") == DatabaseMetaData.columnNullable);

                pks.add(new PKMetaDataHolder(dataType, typeName, columnSize, isNullable));
            }
        }
    }
});
