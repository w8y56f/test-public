private Map<String, Map<String, ColumnMetaDataHolder>> queryMetadata(List<TableNameEnt> rule99List, Connection bsConn,
                                                                      Connection nbBffConn, Connection uwBffConn, Connection claimBsConn) throws Exception {

    // 1. 使用 supplyAsync 代替 runAsync，以便捕获返回值
    List<CompletableFuture<Map<String, Map<String, ColumnMetaDataHolder>>>> futures = rule99List.stream()
            .map(table -> CompletableFuture.supplyAsync(() -> {
                try {
                    if (table.getTableSource().equalsIgnoreCase("BS")) {
                        return fetchAllColumnsMetadata(bsConn.getMetaData(), table);
                    } else if (table.getTableSource().equalsIgnoreCase("NB-BFF")) {
                        return fetchAllColumnsMetadata(nbBffConn.getMetaData(), table);
                    } else if (table.getTableSource().equalsIgnoreCase("UW-BFF")) {
                        return fetchAllColumnsMetadata(uwBffConn.getMetaData(), table);
                    } else if (table.getTableSource().equalsIgnoreCase("CLAIM-BS")) {
                        return fetchAllColumnsMetadata(claimBsConn.getMetaData(), table);
                    } else {
                        throw new RuntimeException("do not support this table_source:" + table.getTableSource());
                    }
                } catch (SQLException e) {
                    throw new RuntimeException(e);
                }
            }, metadataPool))
            .collect(Collectors.toList());

    // 2. 等待所有任务完成
    try {
        CompletableFuture.allOf(futures.toArray(new CompletableFuture[0])).join();
    } catch (CompletionException e) {
        throw (RuntimeException) e.getCause();
    }

    // 3. 汇总合并所有的结果到一个大的 Map 中返回
    Map<String, Map<String, ColumnMetaDataHolder>> finalResult = new HashMap<>();
    for (CompletableFuture<Map<String, Map<String, ColumnMetaDataHolder>>> future : futures) {
        // 这里的 .join() 不会阻塞，因为前面已经 allOf(...).join() 确保全部完成了
        Map<String, Map<String, ColumnMetaDataHolder>> taskResult = future.join();
        if (taskResult != null) {
            finalResult.putAll(taskResult);
        }
    }

    return finalResult;
}
