
@Override
public ResultVO<QueryOutput> query(QueryInput input) {
    try {
        QueryInput.verify(input);
        String tableName = input.getTableName();

        // 核心修复 1: 必须使用白名单过滤表名
        if (!TABLE_WHITE_LIST.contains(tableName)) {
             throw new IllegalArgumentException("Invalid table name");
        }

        // 核心修复 2: 严格校验列名
        List<String> columnsToMask = input.getConfigColumnList();
        for (String col : columnsToMask) {
            if (!isValidSchema(col)) {
                throw new IllegalArgumentException("Invalid column name: " + col);
            }
        }

        // 获取主键等逻辑...
        List<String> pkListOrdered = queryPkFromDbMetadata(tableName);
        List<String> selectColumnList = Stream.concat(pkListOrdered.stream(), columnsToMask.stream()).toList();
        
        // 使用标准的引号包裹标识符（防止特殊字符注入，根据数据库类型如 Oracle 使用 "，MySQL 使用 `）
        String safeColumns = selectColumnList.stream()
                .map(c -> "\"" + c + "\"") 
                .collect(Collectors.joining(","));
        String safeTable = "\"" + tableName + "\"";

        Map<String, Object> paramMap = new HashMap<>();
        String whereCondition = constructWhereCondition(input, paramMap);

        // 构建 SQL
        String sql = String.format("SELECT %s FROM %s WHERE %s", safeColumns, safeTable, whereCondition);

        Query query = entityManager.createNativeQuery(sql);
        
        // 核心修复 3: 确保所有值都通过 setParameter 绑定
        paramMap.forEach(query::setParameter);

        List<?> rawResultList = query.getResultList();
        // ... 后续逻辑
    } catch (Exception e) {
        // ... 错误处理
    }
}
